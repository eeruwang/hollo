    // We number figures with a script since CSS counters don't work,
    // likely because of the distill template.
    function numberFigures() {
        // Find all figcaption elements with class 'text-caption'
        const figcaptions = document.querySelectorAll('figcaption.text-caption');

        figcaptions.forEach((figcaption, index) => {
            // Create a new span for the figure number
            figcaption.style.marginBottom = '1.25rem';

            const newSpan = document.createElement('span');
            newSpan.style.fontWeight = '500';
            newSpan.textContent = `Figure ${index + 1}: `;

            // Insert the new span at the beginning of the figcaption
            figcaption.insertBefore(newSpan, figcaption.firstChild);
        });
    }

    // Function to create a slug from header text
    function createSlug(text) {
        return text
            .toLowerCase()
            .replace(/[^\w\s]/g, '') // Remove special characters
            .replace(/\s+/g, '-')    // Replace spaces with hyphens
            .trim();                 // Remove whitespace from start/end
    }

    // Function to add IDs to all headers and create a table of contents
    function addHeaderSlugs() {
        // Get all heading elements (h1 through h6)
        const headings = document.querySelectorAll('h1, h2, h3, h4, h5, h6');

        // Process each heading
        headings.forEach((heading) => {
            // Skip headings that already have IDs
            if (heading.id) return;

            // Create slug from heading text
            const slug = createSlug(heading.textContent);

            // Add ID to the heading
            heading.id = slug;
        });
    }

    // Run the function when the document is loaded
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', numberFigures);
        document.addEventListener('DOMContentLoaded', addHeaderSlugs);
    } else {
        // Document already loaded, run the function
        numberFigures();
        addHeaderSlugs();
    }

<script>
    document.addEventListener('DOMContentLoaded', function () {
        // Remove whitespace text nodes between <p> elements in code blocks
        // so that only block breaks contribute spacing (not pre-wrap newlines)
        document.querySelectorAll('pre.code-block-wrap code').forEach(function (code) {
            Array.from(code.childNodes).forEach(function (node) {
                if (node.nodeType === 3 && !node.nodeValue.trim()) node.remove();
            });
        });

        document.querySelectorAll('.tabbed-details').forEach(function (wrapper, wrapperIdx) {
            var details = wrapper.querySelectorAll('details');
            if (details.length === 0) return;

            var container = document.createElement('div');
            container.className = 'tabbed-details-container';

            var nav = document.createElement('div');
            nav.className = 'tabbed-details-nav';
            container.appendChild(nav);

            details.forEach(function (detail, i) {
                var summary = detail.querySelector('summary');
                var label = summary ? summary.textContent : 'Tab ' + (i + 1);

                var btn = document.createElement('button');
                btn.textContent = label;
                btn.dataset.index = i;
                btn.dataset.wrapper = wrapperIdx;
                if (i === 0) btn.className = 'active';
                btn.addEventListener('click', function () {
                    var w = this.dataset.wrapper;
                    var idx = this.dataset.index;
                    var cont = this.closest('.tabbed-details-container');
                    cont.querySelectorAll('.tabbed-details-panel').forEach(function (p) { p.classList.remove('active'); });
                    cont.querySelectorAll('.tabbed-details-nav button').forEach(function (b) { b.classList.remove('active'); });
                    cont.querySelectorAll('.tabbed-details-panel')[idx].classList.add('active');
                    this.classList.add('active');
                });
                nav.appendChild(btn);

                var panel = document.createElement('div');
                panel.className = 'tabbed-details-panel' + (i === 0 ? ' active' : '');
                // Copy all children except the summary
                Array.from(detail.children).forEach(function (child) {
                    if (child.tagName !== 'SUMMARY') {
                        panel.appendChild(child.cloneNode(true));
                    }
                });
                container.appendChild(panel);
            });

            wrapper.parentNode.replaceChild(container, wrapper);
        });

        // Transform inline transcript divs into chat bubbles
        document.querySelectorAll('.transcript-inline').forEach(function (wrapper) {
            var entries = wrapper.querySelectorAll('.t');
            if (entries.length === 0) return;

            var container = document.createElement('div');
            container.className = 'transcript-inline-rendered';

            entries.forEach(function (entry) {
                var role = entry.getAttribute('data-role') || 'agent';
                var name = entry.getAttribute('data-name') || '';

                var bubble = document.createElement('div');
                bubble.className = 't-bubble t-bubble-' + role;

                var content = document.createElement('div');
                content.className = 't-bubble-content';

                if (name) {
                    var label = document.createElement('div');
                    label.className = 't-bubble-label';
                    label.textContent = name;
                    content.appendChild(label);
                }

                var body = document.createElement('div');
                body.innerHTML = entry.innerHTML;
                content.appendChild(body);

                bubble.appendChild(content);
                container.appendChild(bubble);
            });

            wrapper.parentNode.replaceChild(container, wrapper);
        });

        // Basic syntax highlighting for code blocks inside transcript bubbles
        function highlightCode(text, lang) {
            // HTML-escape first
            var escaped = text.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
            // Order matters: comments and strings first (greedy), then keywords etc.
            var rules = [
                // Comments
                { re: /(#.*$)/gm, cls: 'tok-cmt' },
                // Triple-quoted strings
                { re: /("""[\s\S]*?"""|'''[\s\S]*?''')/g, cls: 'tok-str' },
                // Double-quoted strings
                { re: /("(?:[^"\\]|\\.)*")/g, cls: 'tok-str' },
                // Single-quoted strings
                { re: /('(?:[^'\\]|\\.)*')/g, cls: 'tok-str' },
                // Numbers
                { re: /\b(\d+\.?\d*)\b/g, cls: 'tok-num' },
                // Python keywords
                { re: /\b(def|class|if|elif|else|for|while|return|import|from|as|try|except|finally|with|yield|lambda|pass|break|continue|raise|in|not|and|or|is|None|True|False|self|async|await|print)\b/g, cls: 'tok-kw' },
                // Function calls
                { re: /\b([a-zA-Z_]\w*)\s*(?=\()/g, cls: 'tok-fn' }
            ];
            // Apply rules sequentially, protecting already-wrapped spans
            var placeholder = [];
            rules.forEach(function (rule) {
                escaped = escaped.replace(rule.re, function (match, g1) {
                    var idx = placeholder.length;
                    placeholder.push('<span class="' + rule.cls + '">' + g1 + '</span>');
                    return '\x00_' + idx + '_\x01';
                });
            });
            // Restore placeholders
            placeholder.forEach(function (span, i) {
                escaped = escaped.replace('\x00_' + i + '_\x01', span);
            });
            return escaped;
        }

        // Fix code blocks inside transcript bubbles: merge <p> elements into
        // a single preformatted text block so indentation is preserved
        document.querySelectorAll('.transcript-inline-rendered .t-code').forEach(function (block) {
            var lines = [];
            Array.from(block.childNodes).forEach(function (child) {
                if (child.nodeType === 1 && child.tagName === 'P') {
                    lines.push(child.textContent);
                } else if (child.nodeType === 3 && child.textContent.trim()) {
                    lines.push(child.textContent.trim());
                }
            });
            if (lines.length > 0) {
                var text = lines.join('\n');
                var lang = block.getAttribute('data-lang') || 'python';
                block.innerHTML = highlightCode(text, lang);
            }
        });
    });
